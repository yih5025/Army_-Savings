public class MyClass {

public static void main(String args[]) {


double firstprincipal=Double.parseDouble(args[0]);
double secondprincipal=Double.parseDouble(args[1]);
double month=Double.parseDouble(args[2]);
double monthLast=Double.parseDouble(args[3]);
double interest_rate=Double.parseDouble(args[4]);
double BankAccountBalance=Double.parseDouble(args[5]);

System.out.println("First Principal :"+firstprincipal);
System.out.println("Second principal :"+secondprincipal);
System.out.println("Month :"+month);
System.out.println("Last Month :"+monthLast);
System.out.println("Interest Rate :"+interest_rate);
System.out.println("Bank Account Balance :"+BankAccountBalance);

double total_principal = firstprincipal*month + secondprincipal*(month-monthLast);
double total_interest_rate = total_principal * interest_rate * 1/100;
double total_money = total_principal + total_interest_rate + BankAccountBalance;

System.out.println("Total Principal is :"+total_principal);
System.out.println("Total Interest Rate is :"+total_interest_rate);
System.out.println("I can got total money is :"+total_money);

}

}
